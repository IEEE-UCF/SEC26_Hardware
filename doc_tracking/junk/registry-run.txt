docker run -d \
  -p 5000:5000 \
  --restart=always \
  --name registry \
  -v /data/registry:/var/lib/registry \
  -v /data/auth:/auth \
  -v /etc/letsencrypt:/etc/letsencrypt \
  -e "REGISTRY_AUTH=htpasswd" \
  -e "REGISTRY_AUTH_HTPASSWD_REALM=Registry Realm" \
  -e "REGISTRY_AUTH_HTPASSWD_PATH=/auth/htpasswd" \
  -e REGISTRY_HTTP_TLS_CERTIFICATE=/etc/letsencrypt/live/registry.syndric.org/fullchain.pem \
  -e REGISTRY_HTTP_TLS_KEY=/etc/letsencrypt/live/registry.syndric.org/privkey.pem \
  registry:2
